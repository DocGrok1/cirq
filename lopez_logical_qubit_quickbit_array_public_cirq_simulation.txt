import cirq
import numpy as np
import matplotlib.pyplot as plt


def build_public_lopez_demo_circuit():
    """
    Build a small public Cirq demonstration circuit.

    This circuit is a simplified public behavior model only.
    It is not the full protected Lopez Logical Qubit / Quickbit Array system.
    """

    # Public demonstration register.
    q0, q1, q2 = cirq.LineQubit.range(3)

    circuit = cirq.Circuit()

    # 1. Initialize public coherent state.
    circuit.append(cirq.H(q0))

    # 2. Entangle support qubits.
    circuit.append(cirq.CNOT(q0, q1))
    circuit.append(cirq.CNOT(q1, q2))

    # 3. Apply bounded reversible public transformation layer.
    # These gates are illustrative placeholders for a public simulation layer.
    # They are not the protected internal control rules.
    circuit.append(cirq.Z(q0) ** 0.25)
    circuit.append(cirq.X(q1) ** 0.5)

    # 4. Feedback-style controlled operation.
    # This demonstrates a reversible dependency path in the public model.
    circuit.append(cirq.CNOT(q2, q0))

    # 5. Measure all public demonstration qubits.
    circuit.append(cirq.measure(q0, q1, q2, key="result"))

    return circuit


def run_public_lopez_demo(repetitions=1000, seed=9):
    """
    Run the public demonstration simulation.

    Args:
        repetitions: Number of measurement repetitions.
        seed: Fixed simulator seed for reproducible public output.

    Returns:
        circuit: The Cirq circuit.
        result: The Cirq simulation result.
        histogram: Measurement histogram.
    """

    circuit = build_public_lopez_demo_circuit()
    simulator = cirq.Simulator(seed=seed)
    result = simulator.run(circuit, repetitions=repetitions)
    histogram = result.histogram(key="result")

    return circuit, result, histogram


if __name__ == "__main__":
    circuit, result, histogram = run_public_lopez_demo(
        repetitions=1000,
